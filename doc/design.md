### 💡 类API
```cangjie

// enum API
public enum B {
    | case1
    | case2
}

// struct API
public struct S {
    public func aa(): Unit
    public func xx(): Unit
    public func cc(): Unit
}

// class API
public class AA {
    public func aa(): Unit 
    public func bb(): String 
    public func cc(arg: Int64): Unit 
    public func dd(arg: Bool): Unit 
    public static func ee(): Unit 
    public static func ff(arg: String): Unit 
}

// function API
public func abc () : Unit
public func def () : Bool

```
